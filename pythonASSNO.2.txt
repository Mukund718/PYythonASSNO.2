    - (5 > 4) or (3 == 5)

    - (True and True) and (True == False)

- not (5 > 4) -> not(True) = False
- (5 > 4) or (3 == 5) -> True or False = True
- not ((5 > 4) or (3 == 5)) -> not(True) - >False
